배열 메소드 

메소드 method :  (객체의)동작=기능=function

            let fruits=["Banana", "Apple", "Orange", "Mango"];
            
            document.write(fruits);                    //배열의 전체 요소 출력 : Banana,Apple,Orange,Mango
            document.write("<hr>");
            document.write(fruits.toString());         //배열 요소를 문자열로 변경 : Banana,Apple,Orange,Mango
            document.write("<hr>");
            document.write("#"+fruits.toString()+"#"); //#Banana,Apple,Orange,Mango#
            document.write("<hr>");
            
            document.write(fruits.length);             //배열 요소의 개수 : 4
            document.write("<hr>");
            document.write(fruits.toString().length);  //배열 요소를 하나의 큰 문자열 변환 후 그것의 글자 개수
                                                       //: 25 (글자수 출력됨)
                                                       
                                                       
            document.write(fruits.join("*"));          //요소와 요소를 ()안에 있는것으로 연결
            document.write("<hr>");                    //Banana*Apple*Orange*Mango
            
            fruits.pop()                               // 배열의 마지막 요소 제거 
            document.write(fruits);                    //Banana,Apple,Orange
            document.write("<hr>");  

            fruits.push("kiwi");                       // 배열의 마지막 요소 추가
            document.write(fruits);                    //Banana,Apple,Orange,kiwi
            document.write("<hr>");
            
            fruits.shift();                            //배열 맨앞 요소 제거
            document.write(fruits);                    //Apple,Orange,kiwi
            document.write("<hr>");

            fruits.unshift("Lemon");                   //배열 맨앞 요소 추가
            document.write(fruits);                    //Lemon,Apple,Orange,kiwi
            document.write("<hr>");

정렬

어떤 Key값을 기준으로 순서대로 재배치하는 것 
정렬유형 :
1)오름차순 ASC (Ascending) : 1->10, A->Z, A(대문자)->a(소문자)
2)내림차순 DESC (Descending) : 10->1
정렬방식 : insertion sort(삽입정렬), selection sort(선택 정렬), bubble sort(버블 정렬),...


1)문자형 데이터 정렬

fruits = ["Banana","Orange","Apple","Mango"];

fruits.sort();                  //배열 요소를 오름차순 정렬
document.write(fruits);         //Apple,Banana,Mango,Orange
document.write("<hr>");

fruits.reverse();               //배열 요소를 내림차순 정렬
document.write(fruits);         //Orange,Mango,Banana,Apple
document.write("<hr>");

fruits.sort();  : 배열 요소를 오름차순 정렬 

fruits.reverse();     : 배열 요소를 내림차순 정렬



2)숫자형 데이터 정렬 Numeric sort


points = [40, 100, 1, 5, 25, 10];

points.sort(function(a, b){return a - b});                 
document.write(points);         //오름차순 : 1,5,10,25,40,100
document.write("<hr>");

points.sort(function(a, b){return b - a});             
document.write(points);         //내림차순 : 100,40,25,10,5,1
document.write("<hr>");


sort()는 문자열 정렬에선 잘 작동하지만 숫자데이터는 문자열처럼 정렬하여 잘 작동하지못함(ex:앞자리 2,1때문에 25가 100보다 큰 것으로 인식됨)
즉  숫자데이터에선 sort()가 잘못된 결과를 리턴함 -> 비교함수(compare function)로 문제해결가능

points.sort(function(a, b){return a - b}); 

points.sort(function(a, b){return b - a});
